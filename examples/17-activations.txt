// Example 17: Activations
// Demonstrates activate, deactivate, autoactivation, and activecolor

title Activations Demo

participant Client
participant Server
participant Worker

// Manual activation
activate Client
Client->Server:Start processing

activate Server
Server->Worker:Delegate task

activate Worker
Worker-->Server:Task complete
deactivate Worker

Server-->Client:Processing done
deactivate Server

Client->Client:Update UI
deactivate Client

// Auto-activation mode
autoactivation on

Client->Server:Auto-activated request
Server->Worker:Auto-activated task
Worker-->Server:Done
Server-->Client:Response

autoactivation off

// Colored activations
activate Client #lightblue
Client->Server:Blue activation
deactivate Client

// activecolor directive
activecolor #lightgreen

activate Server
Server->Worker:Green activation
deactivate Server
