// Example 5: Styled Fragments
// Demonstrates fragment styling with colors and borders

title Order Processing with Styled Fragments

participant Client
participant OrderService
database Inventory

// Fragment with operator color (yellow label corner)
alt#yellow order valid
  OrderService->Inventory:Check stock
  Inventory-->OrderService:In stock
  OrderService-->Client:Order confirmed
else#red out of stock
  OrderService-->Client:Item unavailable
end

// Fragment with fill color (light blue background)
loop #lightblue processing queue
  OrderService->Inventory:Process next
  Inventory-->OrderService:Done
end

// Fragment with full styling (yellow label, pink fill, red border)
alt#yellow #pink #red;2;dashed premium order
  OrderService->Client:Priority handling
else #lightgray standard order
  OrderService->Client:Normal handling
end
