// Example 16: Lifeline Styling
// Demonstrates lifelinestyle directive for customizing lifelines

title Lifeline Styling Demo

participant Client #lightblue
participant Server #lightgreen
participant Database #lightyellow

// Default lifeline style for all
lifelinestyle #gray;1;dashed

// Override for specific participant
lifelinestyle Client #blue;2;solid
lifelinestyle Database #red;2;dotted

Client->Server:Request
Server->Database:Query

// Change mid-diagram
lifelinestyle Server #green;3;solid

Database-->Server:Results
Server-->Client:Response

// Additional messages show updated styles
Client->Server:Another request
Server->Database:Another query
Database-->Server:More results
Server-->Client:Final response
