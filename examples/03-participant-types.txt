// Example 3: Participant Types and Styling
// Demonstrates participant, actor, database with styling

title Participant Types

// Regular participant (box) with blue fill
participant "API Gateway" as API #lightblue

// Actor (stick figure) with yellow fill
actor "End User" as User #yellow

// Database (cylinder) with green styling
database "PostgreSQL" as DB #lightgreen #darkgreen;2

// Standard participant with border styling
participant "Cache" as Cache #white #navy;2;dashed

// Flow between different types
User->API:Request
API->Cache:Check cache
Cache-->API:Cache miss
API->DB:Query
DB-->API:Data
API-->User:Response
