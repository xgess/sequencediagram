// Example 20: Participant Groups
// Demonstrates participantgroup for visual grouping

title Participant Groups Demo

// Group participants with a label
participantgroup #lightblue Frontend
  participant Browser
  participant Mobile
end

participantgroup #lightgreen Backend
  participant API
  participant Service
end

participantgroup #lightyellow Data
  database Cache
  database DB
end

Browser->API:HTTP Request
Mobile->API:API Request

API->Service:Process
Service->Cache:Check cache

alt Cache hit
  Cache-->Service:Data
else Cache miss
  Service->DB:Query
  DB-->Service:Data
  Service->Cache:Store
end

Service-->API:Result
API-->Browser:Response
API-->Mobile:Response
